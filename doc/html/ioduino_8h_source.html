00160 void IOduino::init(std::string filename) {
00161         if(filename == "") { // If no filename given, then cycle through /dev/ttyUSB0->9
00162                 for(int i = 0; i < 10; i++) {
00163                         std::stringstream filename_stream;
00164                         filename_stream << "/dev/ttyUSB" << i;
00165                         if (ConnectToArduino(filename_stream.str()) == true)
00166                                 break;
00167                 }
00168                 if(!isConnected())
00169                         throw IOduino_exception("Unable to connect to IOduino at ports /dev/ttyUSB[0-9]");
00170         } else { 
00171                 ConnectToArduino(filename);
00172                 if(!isConnected()) 
00173                         throw IOduino_exception("Unable to connect to IOduino at port " + filename);
00174         }
00175 }

00177 bool IOduino::ConnectToArduino(std::string filename) {
00178         IOduino_stream.Open(filename);
00179         if(IOduino_stream.IsOpen()) { // Found a USB->serial device
00180                 IOduino_stream.SetBaudRate(BAUD_RATE);
00181                 IOduino_stream.SetCharSize( LibSerial::SerialStreamBuf::CHAR_SIZE_8 ) ;
00182                 IOduino_stream.SetNumOfStopBits(1);
00183                 IOduino_stream.SetParity(LibSerial::SerialStreamBuf::PARITY_NONE);
00184                 IOduino_stream.SetFlowControl(LibSerial::SerialStreamBuf::FLOW_CONTROL_NONE);
00185 
00186                 IOduino_stream << 'v'; // Verify it is an Arduino with the IOduino firmware
00187                                        // It should respond with an 'i'
00188                 char response;
00189                 IOduino_stream >> response;
00190                 if (response == 'i') {
00191                         return true;
00192                 } else {
00193                         IOduino_stream.Close();
00194                         return false;
00195                 }
00196         } else { // Couldn't open file
00197                 return false;
00198         }
00199 };
